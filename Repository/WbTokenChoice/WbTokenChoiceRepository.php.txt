declare(strict_types=1);

namespace BaksDev\Wildberries\Repository\WbTokenChoice;

use BaksDev\Core\Doctrine\DBALQueryBuilder;
use BaksDev\Users\Profile\UserProfile\Entity\Event\Info\UserProfileInfo;
use BaksDev\Users\Profile\UserProfile\Entity\Event\Personal\UserProfilePersonal;
use BaksDev\Users\Profile\UserProfile\Entity\UserProfile;
use BaksDev\Users\Profile\UserProfile\Type\Id\UserProfileUid;
use BaksDev\Users\Profile\UserProfile\Type\UserProfileStatus\Status\UserProfileStatusActive;
use BaksDev\Users\Profile\UserProfile\Type\UserProfileStatus\UserProfileStatus;
use BaksDev\Wildberries\Entity\Event\Profile\WbTokenProfile;
use BaksDev\Wildberries\Entity\WbToken;
use BaksDev\Wildberries\Repository\WbTokenByProfile\WbTokenByProfileInterface;
use Generator;

final readonly class WbTokenChoiceRepository implements WbTokenChoiceInterface
{

    public function __construct(
        private WbTokenByProfileInterface $wbTokenByProfile,
        private DBALQueryBuilder $DBALQueryBuilder
    ) {}

    /**
     * Возвращает всю коллекцию идентификаторов токенов
     *
     * @return Generator<UserProfileUid>|false
     */
    public function getTokenCollection(): Generator|false
    {
        $dbal = $this->DBALQueryBuilder->createQueryBuilder(self::class);

        $dbal->from(WbToken::class, 'token');

        $dbal->join(
            'token',
            WbTokenProfile::class,
            'wb_token_profile',
            'wb_token_profile.event = token.event',
        );

        $dbal
            ->join(
                'wb_token_profile',
                UserProfileInfo::class,
                'users_profile_info',
                'users_profile_info.profile = wb_token_profile.value AND users_profile_info.status = :status',
            )
            ->setParameter(
                key: 'status',
                value: UserProfileStatusActive::class,
                type: UserProfileStatus::TYPE,
            );

        $dbal->leftJoin(
            'wb_token_profile',
            UserProfile::class,
            'users_profile',
            'users_profile.id = wb_token_profile.value',
        );

        $dbal->leftJoin(
            'users_profile',
            UserProfilePersonal::class,
            'users_profile_personal',
            'users_profile_personal.event = users_profile.event',
        );


        //        $select = sprintf('new %s(token.id, users_profile_personal.username)', UserProfileUid::class);
        //        $dbal->select($select);

        $dbal
            ->addSelect('wb_token_profile.value AS value')
            ->addSelect('users_profile_personal.username AS attr');

        /* Кешируем результат ORM */
        return $dbal
            ->enableCache('wildberries', '1 day')
            ->fetchAllHydrate(UserProfileUid::class);
    }

    /**
     * Возвращает коллекцию идентификаторов, доступных активному профилю пользователя
     *
     * @return Generator<UserProfileUid>|false
     */
    public function getAccessProfileTokenCollection(): Generator|false
    {
        $UserProfileUid = $this->wbTokenByProfile->getCurrentUserProfile();

        if(false === ($UserProfileUid instanceof UserProfileUid))
        {
            return false;
        }

        $dbal = $this->DBALQueryBuilder->createQueryBuilder(self::class);


        $dbal->from(WbToken::class, 'token');

        $dbal->join(
            'token',
            WbTokenProfile::class,
            'wb_token_profile',
            'wb_token_profile.event = token.event',
        );

        /*$dbal->join(
            WbTokenAccess::class,
            'access',
            'WITH',
            'access.event = token.event AND access.profile = :access',
        );*/


        //$dbal->setParameter('access', $UserProfileUid, UserProfileUid::TYPE);


        $dbal
            ->join(
                'wb_token_profile',
                UserProfileInfo::class,
                'users_profile_info',
                'users_profile_info.profile = wb_token_profile.value AND users_profile_info.status = :status',
            )
            ->setParameter(
                'status',
                UserProfileStatusActive::class,
                UserProfileStatus::TYPE,
            );

        $dbal->leftJoin(
            'wb_token_profile',
            UserProfile::class,
            'users_profile',
            'users_profile.id = wb_token_profile.value',
        );

        $dbal->leftJoin(
            'users_profile',
            UserProfilePersonal::class,
            'users_profile_personal',
            'users_profile_personal.event = users_profile.event',
        );


        //$select = sprintf('new %s(token.id, users_profile_personal.username)', UserProfileUid::class);
        //$dbal->select($select);

        $dbal
            ->addSelect('wb_token_profile.value AS value')
            ->addSelect('users_profile_personal.username AS attr');

        /* Кешируем результат ORM */
        return $dbal
            ->enableCache('wildberries', '1 day')
            ->fetchAllHydrate(UserProfileUid::class);

    }
}
