namespace BaksDev\Wildberries\Api\Token\Reference\Ğ¡haracteristics\ConfigCard;

final class PropertyDTO
{
    private bool $available;
    private bool $required;
    private bool $onlyDictionary = false;
    private string $type;
    private ?string $dictionary = null;
    private ?int $maxCount = null;
    private bool $isNumber;
    private ?string $units = null;
    

    public function __construct(
      array $data
    )
    {
        $this->required = $data['required'];
        $this->available = $data['popular'];
        $this->type = $data['name'];
        $this->maxCount = $data['maxCount'] ?? null;
        $this->units = $data['unitName'];
        $this->isNumber = $data['charcType'] === 4;
        
        //$this->onlyDictionary = $data['useOnlyDictionaryValues'];
        
        //$this->dictionary = $data['dictionary'] ?? null;
    }
    
    /**
     * @return bool
     */
    public function isAvailable() : bool
    {
        return $this->available;
    }
    
    /**
     * @return bool
     */
    public function isRequired() : bool
    {
        return $this->required;
    }
    
    /**
     * @return bool
     */
    public function isOnlyDictionary() : bool
    {
        return $this->onlyDictionary;
    }
    
    /**
     * @return string
     */
    public function getType() : string
    {
        return $this->type;
    }
    
    /**
     * @return string|null
     */
    public function getDictionary() : ?string
    {
        return $this->dictionary;
    }
    
    /**
     * @return int|null
     */
    public function getMaxCount() : ?int
    {
        return $this->maxCount;
    }
    
    /**
     * @return bool
     */
    public function isNumber() : bool
    {
        return $this->isNumber;
    }
    
    /**
     * @return int|null
     */
    public function getUnits() : ?string
    {
        return $this->units;
    }
    
}
