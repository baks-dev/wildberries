final class VariationDTO
{
    private bool $available; //": true,
    private bool $required; //": true,
    private bool $onlyDictionary = false; //": true,
    private string $type; //": "Размер",
    private ?string $dictionary = null; //": "/tech-sizes"
    
    /**
     * @param bool $isAvailable
     * @param bool $required
     * @param bool $useOnlyDictionaryValues
     * @param string $type
     * @param string|null $dictionary
     */
    public function __construct(array $data)
    {
    
        $this->required = $data['required'];
        $this->available = $data['popular'];
        $this->type = $data['name'];
        $this->maxCount = $data['maxCount'] ?? null;
        $this->units = $data['unitName'];
        $this->isNumber = $data['charcType'] === 4;
        
        
//        $this->available = $data['isAvailable'];
//        $this->required = $data['required'];
//        $this->onlyDictionary = $data['useOnlyDictionaryValues'];
//        $this->type = $data['type'];
//        $this->dictionary = $data['dictionary'] ?? null;
    }
    
    /**
     * @return bool
     */
    public function isAvailable() : bool
    {
        return $this->available;
    }
    
    /**
     * @return bool
     */
    public function isRequired() : bool
    {
        return $this->required;
    }
    
    /**
     * @return bool
     */
    public function isOnlyDictionary() : bool
    {
        return $this->onlyDictionary;
    }
    
    /**
     * @return string
     */
    public function getType() : string
    {
        return $this->type;
    }
    
    /**
     * @return string|null
     */
    public function getDictionary() : ?string
    {
        return $this->dictionary;
    }
    
}
